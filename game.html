function draw() {
  background(0);

  var currentTime = millis() - startTime;

  // Spawn notes
  for (var i = 0; i < noteChart.length; i++) {
    var n = noteChart[i];
    if (!n.spawned && currentTime >= n.time) {
      notes.push(new Note(n.lane));
      n.spawned = true;
    }
  }

  // Hit line
  stroke(255);
  line(0, hitLineY, width, hitLineY);

  // Notes update
  for (var i = notes.length - 1; i >= 0; i--) {
    notes[i].update();
    notes[i].show();

    if (notes[i].y > height) {
      notes.splice(i, 1);
      health -= 10;
    }
  }

  fill(255);
  textSize(20);
  text("Score: " + score, 10, 20);
  text("Health: " + health, 10, 50);

  if (health <= 0) {
    song.stop();
    noLoop();
    textSize(40);
    text("Game Over", 200, 250);
  }

  if (!song.isPlaying() && health > 0) {
    noLoop();
    textSize(40);
    text("You Win!", 200, 250);
  }
}

function keyPressed() {
  var pressed = key.toLowerCase();

  for (var i = notes.length - 1; i >= 0; i--) {
    var n = notes[i];

    if (n.lane === pressed && abs(n.y - hitLineY) < 30) {
      notes.splice(i, 1);
      score += 100;
      return;
    }
  }

  health -= 5;
}

function Note(lane) {
  this.lane = lane;
  this.x = laneX[lane];
  this.y = 0;
  this.speed = 6.5;

  this.update = function () {
    this.y += this.speed;
  };

  this.show = function () {
    fill(255, 0, 0);
    ellipse(this.x, this.y, 30);
  };
}
